rednet.open("right")
local func

term.clear()
term.setCursorPos(1,1)
print("TurtleSpawner program started.")

function fueler()
	if turtle.getFuelLevel() < 400  then
		turtle.select(16)
		turtle.refuel()
	end
end

fueler()
