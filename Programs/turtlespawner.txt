rednet.open("right")
local func

function refuel()
	turtle.select(16)
	turtle.refuel()
	turtle.select(1)
end

refuel()

function netherbook()
	turtle.select(1)
	sleep(0.5)
	turtle.drop()
end

function theendbook()
	turtle.select(2)
	sleep(0.5)
	turtle.drop()
end

function miningagebook()
	turtle.select(3)
	turtle.drop()
end

function oilfieldbook()
	turtle.select(4)
	sleep(0.5)
	turtle.drop()
end

function mback()
	turtle.back()
end

function mforward()
	turtle.forward()
end

while true do
   local id,msg,dist = rednet.receive()
   func = loadstring(msg.."(...)")
   setfenv(func, getfenv())
   func(id)
   rednet.send(id,"Done")
end
