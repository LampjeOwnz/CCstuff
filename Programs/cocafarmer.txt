rednet.open("right")

local turtID = 417
local timer = 300

os.loadAPI("turt")

local lengthA = 7
local lengthB = 8
local width = 4
local layercount = 0

term.clear()
term.setCursorPos(1,1)
print("Coca farmer started")

function goback()
	layercount = 0
	turt.move("forward")
	turt.move("forward")
	turt.move("forward")
	turt.move("left")
	turt.move("forward")
	turt.move("left")
	turt.move("down")
	turt.dropall()
end
	
function layer()
	layercount = layercount + 1
	if layercount < 3 then
		turt.move("up")
		turt.move("left")
		turt.move("forward")
		turt.move("forward")
		turt.move("left")
		cocafarm()
	end
	if layercount == 3 then
		goback()
	end
end

function cocafarm()
	turt.move("right")
	
	for i=1, lengthA do
		if i~=lengthA then
			turt.move("forward")
		end
	end
	
	turt.move("left")
	
	for i=1, width do
		if i~=width then
			turt.move("forward")
		end	
	end
	
	turt.move("left")
	
	for i=1, lengthB do
		if i~=lengthB then
			turt.move("forward")
		end
	end
	
	turt.move("left")
	
	for i=1, width do
		if i~=width then
			turt.move("forward")
		end	
	end
	layer()
end
	

function go()
	print("Starting to farm coca beans")
	turt.move("forward")
	turt.move("down")
	turt.move("forward")
	turt.move("forward")
	rednet.send(turtID, "follow")
	cocafarm()
end

while true do 
	go()
	turt.fuellvl()
	term.clear()
	term.setCursorPos(1,1)
	print("Coca farm is done waiting for next cycle.")
	sleep(timer)
end
